    <script type="text/x-template" id="contact-template">
        <div class="w-full space-y-6">

            <p class="text-sm transition-colors" :class="isDark ? 'text-gray-300' : 'text-gray-600'">Have a question or want to collaborate? Feel free to reach out. I'm always open to discussing new projects and opportunities.</p>
            <form @submit.prevent="handleSubmit" action="https://docs.google.com/forms/u/0/d/e/1FAIpQLSd1VGfIJytEIIcfFz75P_2B_IpBcyCEKO4bPQO1SS0l4GQUOw/formResponse" method="POST" target="hidden-iframe" ref="form" class="space-y-4">

'bg-gray-800 border-gray-700 text-white placeholder-gray-400' : 'bg-gray-100 border-gray-200 text-gray-900 placeholder-gray-500'">

'bg-gray-800 border-gray-700 text-white placeholder-gray-400' : 'bg-gray-100 border-gray-200 text-gray-900 placeholder-gray-500'">

                    <label for="message" class="text-sm font-bold sr-only transition-colors" :class="isDark ? 'text-gray-400' : 'text-gray-600'">Message</label>

'bg-gray-800 border-gray-700 text-white placeholder-gray-400' : 'bg-gray-100 border-gray-200 text-gray-900 placeholder-gray-500'"></textarea>

    <script>
        const { createApp, ref, onMounted, computed, provide, inject } = Vue;

        // Define Components from templates
        const HomeComponent = { 
            template: '#home-template',
            setup() {
                const experienceTab = ref('education'); // Default to 'education'
                const isDark = inject('isDark');
                return { experienceTab, isDark };
            }
        };
        const ProjectsComponent = { 
            template: '#projects-template',
            setup() {
                const isDark = inject('isDark');
                return { isDark };
            }
        };
        const ResumeComponent = { 
            template: '#resume-template',
            setup() {
                const isDark = inject('isDark');
                return { isDark };
            }
        };
        const ContactComponent = {
            template: '#contact-template',
            setup() {
                const showPopup = inject('showPopup');
                const isDark = inject('isDark');
                const form = ref(null);
                const handleSubmit = () => {
                    // Submit the form to the hidden iframe
                    form.value.submit();
                    // Show the popup
                    showPopup.value = true;
                    // Reset the form fields
                    form.value.reset();
                    // Hide the popup after a delay
                    setTimeout(() => { showPopup.value = false; }, 3000);
                };
                return { handleSubmit, form, isDark };
            }
        };

        const app = createApp({
            components: {
                HomeComponent,
                ProjectsComponent,
                ResumeComponent,
                ContactComponent
            },
            setup() {
                // --- STATE ---
                const currentSection = ref('home');
                const portfolioCard = ref(null);
                const cursorDot = ref(null);
                const isDesktop = ref(false);
                const showPopup = ref(false);
                const isDark = ref(false);

                // Provide the state to child components
                provide('showPopup', showPopup);
                provide('isDark', isDark);

                // --- COMPUTED ---
                const currentSectionComponent = computed(() => {
                    switch (currentSection.value) {
                        case 'projects': return ProjectsComponent;
                        case 'resume': return ResumeComponent;
                        case 'contact': return ContactComponent;
                        default: return HomeComponent;
                    }
                });

                // --- METHODS ---
                const switchSection = (section) => {
                    currentSection.value = section;
                };

                const toggleTheme = () => {
                    isDark.value = !isDark.value;
                    document.body.className = isDark.value ? 'dark-theme w-full min-h-screen p-4 sm:p-8 md:p-16 isolate' : 'light-theme w-full min-h-screen p-4 sm:p-8 md:p-16 isolate';
                    localStorage.setItem('theme', isDark.value ? 'dark' : 'light');
                };

                const updateGrid = () => {
                    const gridContainer = document.getElementById('interactive-grid');
                    if (!gridContainer) return;
                    gridContainer.innerHTML = '';
                    const width = window.innerWidth;
                    const height = window.innerHeight;
                    const numColumns = Math.ceil(width / 30);
                    const numRows = Math.ceil(height / 30);
                    const numCells = numColumns * numRows;
                    for (let i = 0; i < numCells; i++) {
                        const cell = document.createElement('div');
                        cell.classList.add('grid-cell');
                        gridContainer.appendChild(cell);
                    }
                };

                const handleCardMouseMove = (e) => {
                    if (!isDesktop.value) return;
                    const card = portfolioCard.value;
                    const rect = card.getBoundingClientRect();
                    const x = e.clientX - rect.left;
                    const y = e.clientY - rect.top;
                    const rotateX = ((y / rect.height) - 0.5) * -15;
                    const rotateY = ((x / rect.width) - 0.5) * 15;
                    card.style.transform = `perspective(1500px) rotateX(${rotateX}deg) rotateY(${rotateY}deg) scale3d(1.05, 1.05, 1.05)`;
                };

                const handleCardMouseEnter = () => {
                    if (!isDesktop.value) return;
                    portfolioCard.value.style.boxShadow = '0px 40px 100px -10px rgba(99, 102, 241, 0.3)';
                };

                const handleCardMouseLeave = () => {
                    if (!isDesktop.value) return;
                    portfolioCard.value.style.transform = 'perspective(1500px) rotateX(0deg) rotateY(0deg) scale3d(1, 1, 1)';
                    portfolioCard.value.style.boxShadow = '0px 20px 50px -10px rgba(0, 0, 0, 0.1)';
                };

                // --- LIFECYCLE HOOKS ---
                onMounted(() => {
                    // Initialize theme from localStorage
                    const savedTheme = localStorage.getItem('theme');
                    if (savedTheme === 'dark') {
                        isDark.value = true;
                        document.body.className = 'dark-theme w-full min-h-screen p-4 sm:p-8 md:p-16 isolate';
                    } else {
                        document.body.className = 'light-theme w-full min-h-screen p-4 sm:p-8 md:p-16 isolate';
                    }

                    isDesktop.value = window.matchMedia('(hover: hover) and (pointer: fine)').matches;
                    
                    updateGrid();
                    window.addEventListener('resize', updateGrid);

                    if (!isDesktop.value) {
                        if (cursorDot.value) cursorDot.value.style.display = 'none';
                        return;
                    }
                    
                    // --- ULTIMATE OPTIMIZED CURSOR LOGIC ---
                    const cursor = cursorDot.value;
                    
                    let mouseX = 0, mouseY = 0;
                    let currentX = 0, currentY = 0;
                    let targetX = 0, targetY = 0;
                    const ease = 0.1;

                    let hoveredNavElement = null;

                    // 1. Listen for mouse movement
                    window.addEventListener('mousemove', e => {
                        mouseX = e.clientX;
                        mouseY = e.clientY;
                    });
                    
                    // 2. Use event delegation to efficiently manage hover states
                    document.addEventListener('mouseover', e => {
                        const target = e.target;
                        
                        // Check for nav link first (highest priority)
                        if (target.closest('.nav-link')) {
                            hoveredNavElement = target.closest('.nav-link');
                            cursor.classList.add('nav-hover');
                        } else if (target.closest('.cursor-interactive')) {
                            const fontSize = parseFloat(window.getComputedStyle(target).fontSize);
                            cursor.style.setProperty('--cursor-height', `${fontSize * 1.2}px`);
                            cursor.classList.add('text-hover');
                        } else if (target.closest('a, button')) {
                            cursor.classList.add('link-hover');
                        }
                    });

                    document.addEventListener('mouseout', e => {
                        hoveredNavElement = null;
                        cursor.className = 'cursor-dot'; // Reset to base class
                    });

                    // 3. The animation loop ONLY handles position
                    const animateCursor = () => {
                        targetX = mouseX;
                        targetY = mouseY;
                        
                        // If hovering a nav link, snap the target position
                        if (hoveredNavElement) {
                            const rect = hoveredNavElement.getBoundingClientRect();
                            targetX = rect.left + rect.width / 2;
                            targetY = rect.top + rect.height / 2;
                            cursor.style.setProperty('--cursor-width', `${rect.width}px`);
                            cursor.style.setProperty('--cursor-height', `${rect.height}px`);
                        } else {
                            // Reset width/height properties when not on a nav link
                            // The actual values will be controlled by the CSS classes
                            cursor.style.removeProperty('--cursor-width');
                            if (!cursor.classList.contains('text-hover')) {
                                cursor.style.removeProperty('--cursor-height');
                            }
                        }

                        // Apply easing for smooth movement
                        currentX += (targetX - currentX) * ease;
                        currentY += (targetY - currentY) * ease;
                        
                        // Apply the final transform
                        cursor.style.transform = `translate(${currentX}px, ${currentY}px) translate(-50%, -50%)`;
                        
                        requestAnimationFrame(animateCursor);
                    };

                    animateCursor();
                });

                return {
                    currentSection,
                    portfolioCard,
                    cursorDot,
                    currentSectionComponent,
                    switchSection,
                    toggleTheme,
                    isDark,
                    handleCardMouseMove,
                    handleCardMouseEnter,
                    handleCardMouseLeave,
                    showPopup,
                };
            }
        });
        
        app.mount('#app');
    </script>
</body>
